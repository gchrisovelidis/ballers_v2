// ─── NAME HELPERS ─────────────────────────────────────────────────────────────
function toTitleCase(str) {
  // Handles Greek + Latin, preserves dots (e.g. "Εμμ.")
  return str.replace(/\S+/g, w => {
    if (w.endsWith('.')) return w.charAt(0).toUpperCase() + w.slice(1).toLowerCase();
    return w.charAt(0).toUpperCase() + w.slice(1).toLowerCase();
  });
}

function formatPlayerName(fullName) {
  // Input: "ΠΑΝΟΥΧΟΣ ΑΡΓ. ΝΙΚΟΛΑΟΣ" or "Ανδρεαδάκης Εμμ. Νικόλαος"
  // Output: { surname: "Πανούχος", firstName: "Νικόλαος" }
  const parts = fullName.trim().split(/\s+/);
  // Surname is always first word
  const surname = toTitleCase(parts[0]);
  // First name is the last word (skip middle abbreviation like "ΑΡΓ.")
  // Filter out abbreviations (words ending with dot or 2-4 chars ending in dot)
  const nameParts = parts.slice(1).filter(p => !p.match(/^[Α-Ωα-ωA-Za-z]{2,4}\.$/));
  const firstName = nameParts.length > 0 ? toTitleCase(nameParts[nameParts.length - 1]) : '';
  return { surname, firstName };
}

// TD layout: [0]=num [1]=date+time [2]=opponent [3]=home/away [4]=category [5]=season [6]=preview [7]=venue
function buildSchedule(doc) {
  const rows = doc.querySelectorAll('table tbody tr');
  let html = '', nextGame = null;
  const now = new Date();

  rows.forEach((tr, i) => {
    const tds = tr.querySelectorAll('td');
    if (tds.length < 5) return;

    const dateTime = tds[1].textContent.trim();   // "09/05/2026 22:00"
    const oppName  = (tds[2].querySelector('a') || tds[2]).textContent.trim();
    const oppSlug  = slugFrom(tds[2]);
    const ha       = tds[3].textContent.trim();    // Home / Away
    const cat      = tds[4].textContent.trim();
    const venue    = tds[7] ? tds[7].textContent.trim() : '';

    // Parse datetime
    let gameDate = null;
    const m = dateTime.match(/(\d{2})\/(\d{2})\/(\d{4})\s+(\d{2}):(\d{2})/);
    if (m) gameDate = new Date(+m[3], +m[2]-1, +m[1], +m[4], +m[5]);

    const isFuture = gameDate && gameDate > now;
    const isNext   = isFuture && !nextGame;
    if (isNext) nextGame = { dateTime, oppName, ha, venue, cat, gameDate };

    const odd   = i%2===0 ? 'row-odd' : '';
    const cls   = isNext ? 'next-row' : odd;
    const logo  = logoImg(oppSlug);
    const mark  = isNext ? ' <span style="color:var(--orange);font-family:var(--mono);font-size:0.6rem;">◀ NEXT</span>' : '';

    html += `<tr class="${cls}">

      <td class="td-date">${ha}</td>
      <td class="td-date">${venue}</td>
      <td class="td-date">${cat}</td>

}

// Basketaki results page: date | opponent(team-meta) | W/L | score | status | category | season | article
function buildResults(doc) {
  const rows = doc.querySelectorAll('table.team-result tbody tr');

function buildRoster(doc, playerStats) {
  const firstTable = doc.querySelector('table');
  if (!firstTable) { loadCSVRoster(playerStats); return; }

  const rows = firstTable.querySelectorAll('tbody tr');
  const basketakiPlayers = [];
  rows.forEach(tr => {
    const tds = tr.querySelectorAll('td');
    if (tds.length < 2) return;
    const fullName = tds[0].textContent.trim();
    if (!fullName || fullName.length < 2) return;
    basketakiPlayers.push(fullName);
  });

  if (!basketakiPlayers.length) { loadCSVRoster(playerStats); return; }

  // Load players.csv to get photo mappings via name_gr column
  fetch('players.csv').then(r => r.text()).then(text => {
    const lines   = text.trim().split(/
?
/);
    const delim   = lines[0].includes(';') ? ';' : ',';
    const headers = lines[0].split(delim).map(h => h.trim().replace(/^﻿/,''));
    const csvRows = lines.slice(1).map(l => {
      const v = l.split(delim).map(x => x.trim());
      const o = {}; headers.forEach((h,i) => o[h] = v[i]||''); return o;
    });

    // Build lookup: name_gr (normalised) → { photo, number, name_en }
    const photoMap = {};
    csvRows.forEach(r => {
      const gr = (r.name_gr || '').trim().toUpperCase();
      if (gr) photoMap[gr] = { photo: r.photo||'', number: r.number||'', name_en: r.name_en||'' };
    });

    let html = '';
    basketakiPlayers.forEach(fullName => {
      // 6: Normalise name — title case, strip middle abbreviation
      const { surname, firstName } = formatPlayerName(fullName);
      const rawSurname = fullName.trim().split(/\s+/)[0]; // for stats lookup key

      // initials from normalised names
      const ini = (surname[0]||'?') + (firstName[0]||'');

      // Lookup by normalised Greek name (uppercase for matching)
      const key     = fullName.trim().toUpperCase();
      const csvData = photoMap[key] || {};
      const photo   = csvData.photo || '';
      const number  = csvData.number || '';

      // 2: ring class
      const ringCls = photo ? 'has-photo' : '';

      let avatarInner;
      if (photo) {
        avatarInner = `<img src="images/${photo}" alt="${surname}" onerror="this.parentElement.innerHTML='${ini}';this.parentElement.classList.remove('has-photo')">`;
      } else {
        avatarInner = ini;
      }

      // Stats lookup using original surname (first word, any case)
      const stats = playerStats[rawSurname] || {};
      const pills = stats.ppg ? `
        <div class="p-pills">
          <span class="p-pill ppg"><span class="p-pill-label">PPG</span>${stats.ppg}</span>
          <span class="p-pill rpg"><span class="p-pill-label">RPG</span>${stats.rpg}</span>
          <span class="p-pill apg"><span class="p-pill-label">APG</span>${stats.apg||'—'}</span>
        </div>` : '';

      html += `<div class="player-tile">
        <div class="p-avatar-wrap">
          <div class="p-avatar ${ringCls}">${avatarInner}</div>
          <div class="p-badge">${number}</div>
        </div>
        <div class="p-surname">${surname}</div>
        <div class="p-name">${firstName}</div>
        ${pills}
      </div>`;
    });

    $('roster-label').textContent = `Active Roster — ${basketakiPlayers.length} Players`;
    $('roster-grid').innerHTML = html;
  }).catch(() => loadCSVRoster(playerStats));
}

function loadCSVRoster(playerStats) {
  fetch('players.csv').then(r => r.text()).then(text => {
    const lines  = text.trim().split(/\r?\n/);
    const delim  = lines[0].includes(';') ? ';' : ',';
    const headers= lines[0].split(delim).map(h => h.trim().replace(/^\uFEFF/,''));
    const rows   = lines.slice(1).map(l => {
      const v = l.split(delim).map(x => x.trim());
      const o = {}; headers.forEach((h,i) => o[h] = v[i]||''); return o;
    }).filter(r => r.name || r.Name);

    let html = '';
    rows.forEach(p => {
      const name   = (p.name||p.Name||'').trim();
      const number = (p.number||p.Number||'').trim();
      const pos    = (p.position||p.Position||'').trim();
      const photo  = (p.photo||p.Photo||'').trim();
      const parts  = name.split(' ');
      const ini    = parts.slice(0,2).map(w=>w[0]||'').join('').toUpperCase();
      const last   = (parts[parts.length-1]||name).toUpperCase();
      const stats  = playerStats[parts[parts.length-1]] || {};
      const statLine = stats.ppg ? `${stats.ppg} PPG · ${stats.rpg} RPG` : '';
      const avatar = photo
        ? `<img src="images/${photo}" alt="${name}" onerror="this.parentElement.innerHTML='${ini}'">`
        : ini;

      html += `<div class="player-tile">
        <div class="p-number">#${number}</div>
        <div class="p-avatar">${avatar}</div>
        <div class="p-name">${last}</div>
        <div class="p-pos">${pos}</div>
        ${statLine ? `<div class="p-stat">${statLine}</div>` : ''}
      </div>`;
    });

    $('roster-label').textContent = `Active Roster — ${rows.length} Players`;
    $('roster-grid').innerHTML = html || `<div style="grid-column:1/-1;text-align:center;padding:2rem;color:var(--muted);font-family:var(--mono);font-size:0.7rem;background:var(--panel);">No players found</div>`;
  }).catch(() => {
    $('roster-grid').innerHTML = `<div style="grid-column:1/-1;text-align:center;padding:2rem;color:var(--muted);font-family:var(--mono);font-size:0.7rem;background:var(--panel);">Could not load roster</div>`;
  });
}

// ─── HEADER & TICKER ─────────────────────────────────────────────────────────
function buildHeader(results) {
  const valid  = results.filter(r => r.ts > 0 || r.os > 0);
  const wins   = valid.filter(r => r.win).length;
  const losses = valid.length - wins;
  const pct    = valid.length ? Math.round(wins/valid.length*100)+'%' : '—';

  $('hdr-w').textContent   = wins;
  $('hdr-l').textContent   = losses;
  $('hdr-pct').textContent = pct;

  // Streak
  let sc = 0, st = 'neutral';
  if (results.length) {
    const first = results[0].win ? 'win' : 'loss';
    st = first;
    for (const r of results) {
      if ((r.win && first==='win')||(!r.win && first==='loss')) sc++;
      else break;
    }
  }
  const chip = $('streak-chip');
  const lbl  = st==='neutral' ? '—' : (st==='win'?'W':'L')+sc;
  chip.textContent = '● Streak '+lbl;
  chip.className   = 'streak-chip '+st;

  // Last result stat cells
  if (results.length) {
    const last = results[0];
    const sv   = $('st-score');
    sv.textContent = `${last.ts}–${last.os}`;
    sv.className   = 'stat-val '+(last.win?'orange':'red');
    $('st-last-opp').textContent = 'vs '+last.oppName;
  }
  const ss = $('st-streak');
  ss.textContent = lbl;
  ss.className   = 'stat-val '+(st==='win'?'orange':st==='loss'?'red':'');
  $('st-streak-sub').textContent = st==='win'?'Hot run 🔥':st==='loss'?'Bounce back time':'—';

  return { wins, losses, pct, st, sc };
}

function startCountdown(target) {
  function tick() {
    const diff = target - new Date();
    if (diff <= 0) { ['cd-d','cd-h','cd-m'].forEach(id => $(id).textContent='00'); return; }
    const d = Math.floor(diff/86400000);
    const h = Math.floor((diff%86400000)/3600000);
    const m = Math.floor((diff%3600000)/60000);
    $('cd-d').textContent = String(d).padStart(2,'0');
    $('cd-h').textContent = String(h).padStart(2,'0');
    $('cd-m').textContent = String(m).padStart(2,'0');
  }
  tick(); setInterval(tick, 30000);
}

// ─── TOGGLE RESULTS ──────────────────────────────────────────────────────────
function toggleResults() {
  const extra = $('results-extra');
  const btn   = $('view-more-btn');
  const row   = $('view-more-row');
  if (!extra) return;
  const isHidden = extra.style.display === 'none';
  extra.style.display = isHidden ? '' : 'none';
  // Move the button row to always be last
  if (isHidden) {
    btn.textContent = '▲ Show Less';
    btn.style.color = 'var(--orange)';
    btn.style.borderColor = 'rgba(255,92,0,0.3)';
    extra.parentNode.appendChild(row);
  } else {
    const total = extra.querySelectorAll('tr').length + 10;
    btn.textContent = `▼ View All ${total} Results`;
    btn.style.color = '#888';
    btn.style.borderColor = 'var(--border2)';
    extra.parentNode.appendChild(row);
  }
}

// ─── MAIN ─────────────────────────────────────────────────────────────────────
async function init() {
  try {
    progress(10, 'Fetching schedule...');
    const schedHTML = await px(`${BASE}/teams/${TEAM}/schedule`);
    const schedDoc  = ph(schedHTML);

    progress(25, 'Parsing schedule...');
    const nextGame = buildSchedule(schedDoc);
    buildNextGameUI(nextGame);

    progress(40, 'Fetching results...');
    const resHTML = await px(`${BASE}/teams/${TEAM}/results`);
    const resDoc  = ph(resHTML);

    progress(55, 'Parsing game log...');
    const results = buildResults(resDoc);
    const hStats  = buildHeader(results);
    buildTicker(hStats, nextGame);

    progress(70, 'Fetching player stats...');
    // Basketaki roster page contains the stats table
    const rosterHTML = await px(`${BASE}/teams/${TEAM}/roster`);
    const rosterDoc  = ph(rosterHTML);

    progress(85, 'Building stats & roster...');
    const playerStats = buildStats(rosterDoc);
    buildRoster(rosterDoc, playerStats);

    progress(100, 'Ready!');
    setTimeout(hideLoader, 400);

  } catch(e) {
    console.error('Init error:', e);
    $('load-text').textContent = 'Error loading data — retrying...';
    setTimeout(hideLoader, 3000);
  }
}

init();
</script>
