// ─── CSV PARSER ───────────────────────────────────────────────────────────────
function parseCSV(text) {
  const lines = text.trim().split(/\r?\n/);
  if (lines.length < 2) return [];
  // detect delimiter
  const delim = lines[0].includes(';') ? ';' : ',';
  const headers = lines[0].split(delim).map(h => h.trim().replace(/^\uFEFF/, ''));
  return lines.slice(1).map(line => {
    const vals = line.split(delim).map(v => v.trim());
    const obj = {};
    headers.forEach((h, i) => obj[h] = vals[i] || '');
    return obj;
  }).filter(row => Object.values(row).some(v => v));
}

// ─── DATE PARSING ─────────────────────────────────────────────────────────────
function parseGameDate(dateStr, timeStr) {
  const formats = [
    { re: /^(\d{1,2})\/(\d{1,2})\/(\d{4})$/, fn: m => new Date(+m[3], +m[2]-1, +m[1]) },
    { re: /^(\d{4})-(\d{2})-(\d{2})$/, fn: m => new Date(+m[1], +m[2]-1, +m[3]) },
  ];
  let d = null;
  for (const f of formats) {
    const m = dateStr.trim().match(f.re);
    if (m) { d = f.fn(m); break; }
  }
  if (!d) return null;
  const [h, min] = (timeStr || '').split(':').map(Number);
  if (!isNaN(h)) { d.setHours(h); d.setMinutes(min || 0); }
  return d;
}

function fmtShortDate(dateStr) {
  const d = parseGameDate(dateStr, '00:00');
  if (!d) return dateStr;
  return d.toLocaleDateString('el-GR', { day:'2-digit', month:'2-digit' }).replace('/', '.');
}

function fmtFullDate(dateStr, timeStr) {
  const d = parseGameDate(dateStr, timeStr || '00:00');
  if (!d) return dateStr;
  const days = ['Sunday','Monday','Tuesday','Wednesday','Thursday','Friday','Saturday'];
  const months = ['January','February','March','April','May','June','July','August','September','October','November','December'];
  return `${days[d.getDay()]}, ${String(d.getDate()).padStart(2,'0')} ${months[d.getMonth()]} ${d.getFullYear()} — ${timeStr || ''}`;
}

// ─── PROGRESS LOADER ──────────────────────────────────────────────────────────
function setProgress(pct, text) {
  document.getElementById('load-bar').style.width = pct + '%';
  document.getElementById('load-text').textContent = text;
}

function hideLoader() {
  const el = document.getElementById('loading');
  el.classList.add('hidden');
  setTimeout(() => el.remove(), 500);
}

// ─── MAIN ─────────────────────────────────────────────────────────────────────
async function init() {
  try {
    setProgress(10, 'Fetching schedule...');
    const [ngRes, resRes, plRes] = await Promise.all([
      fetch('next_game.csv'),
      fetch('results.csv'),
      fetch('players.csv'),
    ]);
    setProgress(50, 'Parsing data...');
    const [ngText, resText, plText] = await Promise.all([
      ngRes.text(), resRes.text(), plRes.text()
    ]);
    const ngRows  = parseCSV(ngText);
    const resRows = parseCSV(resText);
    const plRows  = parseCSV(plText);

    setProgress(70, 'Building game log...');
    buildResults(resRows);

    setProgress(85, 'Loading roster...');
    buildRoster(plRows);

    setProgress(95, 'Setting up scoreboard...');
    buildHeader(resRows);
    buildNextGame(ngRows);
    buildTicker(ngRows, resRows);

    setProgress(100, 'Ready!');
    setTimeout(hideLoader, 400);
  } catch(e) {
    console.error(e);
    document.getElementById('load-text').textContent = 'Error loading data — check console';
  }
}

// ─── RESULTS ──────────────────────────────────────────────────────────────────
function buildResults(rows) {
  // sort newest first by date
  const sorted = rows.slice().sort((a, b) => {
    const da = parseGameDate(a.date || a.Date, '00:00');
    const db = parseGameDate(b.date || b.Date, '00:00');
    return (db || 0) - (da || 0);
  });

  let html = '';
  sorted.forEach((r, i) => {
    const ts  = parseInt(r.team_score || r.TeamScore || 0);
    const os  = parseInt(r.opponent_score || r.OpponentScore || 0);
    const win = ts > os;
    const opp = r.opponent || r.Opponent || '—';
    const dateShort = fmtShortDate(r.date || r.Date || '');
    const score = `${ts}–${os}`;
    const url   = (r.youtube_url || r.YoutubeUrl || '').trim().replace(/&$/, '');
    const watch = url
      ? `<a href="${url}" target="_blank" rel="noopener" class="watch-link">▶ VID</a>`
      : '<span class="no-vid">—</span>';
    const odd = i % 2 === 0 ? 'row-odd' : '';
    html += `
    <tr class="${odd}">
      <td class="td-date">${dateShort}</td>
      <td class="td-opp">vs ${opp}</td>
      <td class="td-score">${score}</td>
      <td class="td-res"><span class="${win ? 'res-w' : 'res-l'}">${win ? 'WIN' : 'LOSS'}</span></td>
      <td class="td-vid">${watch}</td>
    </tr>`;
  });
  document.getElementById('results-body').innerHTML = html || '<tr><td colspan="5" style="text-align:center;color:var(--muted);font-family:var(--mono);font-size:0.7rem;padding:2rem;">No results yet</td></tr>';
}

// ─── HEADER STATS ─────────────────────────────────────────────────────────────
function buildHeader(rows) {
  const wins   = rows.filter(r => parseInt(r.team_score||0) > parseInt(r.opponent_score||0)).length;
  const losses = rows.length - wins;
  const pct    = rows.length ? Math.round(wins / rows.length * 100) + '%' : '—';

  document.getElementById('hdr-wins').textContent   = wins;
  document.getElementById('hdr-losses').textContent = losses;
  document.getElementById('hdr-pct').textContent    = pct;

  // streak
  const sorted = rows.slice().sort((a,b) => {
    const da = parseGameDate(a.date||'', '00:00');
    const db = parseGameDate(b.date||'', '00:00');
    return (db||0)-(da||0);
  });
  let streakCount = 0, streakType = 'neutral';
  if (sorted.length) {
    const first = parseInt(sorted[0].team_score||0) > parseInt(sorted[0].opponent_score||0) ? 'win' : 'loss';
    streakType = first;
    for (const r of sorted) {
      const w = parseInt(r.team_score||0) > parseInt(r.opponent_score||0);
      if ((w && first==='win') || (!w && first==='loss')) streakCount++;
      else break;
    }
  }

  const chip = document.getElementById('streak-chip');
  const label = streakType === 'neutral' ? '—' : (streakType === 'win' ? 'W' : 'L') + streakCount;
  chip.textContent = '● Streak ' + label;
  chip.className = 'streak-chip ' + streakType;

  // stat strip last result
  if (sorted.length) {
    const last = sorted[0];
    const ts = parseInt(last.team_score||0);
    const os = parseInt(last.opponent_score||0);
    const win = ts > os;
    const sv = document.getElementById('st-score');
    sv.textContent = `${ts}–${os}`;
    sv.className = 'stat-val ' + (win ? 'orange' : 'red');
    document.getElementById('st-last-opp').textContent = 'vs ' + (last.opponent || '—');
  }

  // streak stat cell
  const ss = document.getElementById('st-streak');
  const ssLabel = streakType === 'neutral' ? '—' : (streakType === 'win' ? 'W' : 'L') + streakCount;
  ss.textContent = ssLabel;
  ss.className = 'stat-val ' + (streakType==='win' ? 'orange' : streakType==='loss' ? 'red' : '');
  document.getElementById('st-streak-sub').textContent =
    streakType==='win' ? 'Hot run 🔥' : streakType==='loss' ? 'Bounce back time' : '—';
}

// ─── NEXT GAME ────────────────────────────────────────────────────────────────
function buildNextGame(rows) {
  if (!rows.length) return;
  const ng = rows[0];
  const opp  = ng.opponent || ng.Opponent || '—';
  const date = ng.date || ng.Date || '';
  const time = ng.time || ng.Time || '';
  const venue= ng.venue || ng.Venue || '—';
  const ha   = ng.home_away || ng.HomeAway || '—';

  document.getElementById('st-opp').textContent   = opp;
  document.getElementById('st-venue').textContent = `${ha} · ${venue}`;
  document.getElementById('st-time').textContent  = time;
  document.getElementById('st-date').textContent  = fmtFullDate(date, time);

  document.getElementById('cd-opp').textContent   = opp.toUpperCase();
  document.getElementById('cd-date').textContent  = fmtFullDate(date, time);
  document.getElementById('cd-ha').textContent    = ha;
  document.getElementById('cd-venue').textContent = venue;

  const gameDate = parseGameDate(date, time);
  if (gameDate) startCountdown(gameDate);
}

// ─── COUNTDOWN ────────────────────────────────────────────────────────────────
function startCountdown(target) {
  function tick() {
    const diff = target - new Date();
    if (diff <= 0) {
      ['cd-d','cd-h','cd-m'].forEach(id => document.getElementById(id).textContent = '00');
      return;
    }
    const d = Math.floor(diff / 86400000);
    const h = Math.floor((diff % 86400000) / 3600000);
    const m = Math.floor((diff % 3600000) / 60000);
    document.getElementById('cd-d').textContent = String(d).padStart(2,'0');
    document.getElementById('cd-h').textContent = String(h).padStart(2,'0');
    document.getElementById('cd-m').textContent = String(m).padStart(2,'0');
  }
  tick();
  setInterval(tick, 30000);
}

// ─── ROSTER ───────────────────────────────────────────────────────────────────
function buildRoster(rows) {
  document.getElementById('roster-label').textContent = `Active Roster — ${rows.length} Players`;
  let html = '';
  rows.forEach(p => {
    const name   = (p.name || p.Name || '').trim();
    const number = (p.number || p.Number || '').trim();
    const pos    = (p.position || p.Position || '').trim();
    const photo  = (p.photo || p.Photo || '').trim();
    const parts  = name.split(' ');
    const ini    = parts.slice(0,2).map(w => w[0]||'').join('').toUpperCase();
    const lastName = (parts[parts.length-1]||name).toUpperCase();

    const avatarInner = photo
      ? `<img src="images/${photo}" alt="${name}" onerror="this.parentElement.innerHTML='${ini}'">`
      : ini;

    html += `
    <div class="player-tile">
      <div class="p-number">#${number}</div>
      <div class="p-avatar">${avatarInner}</div>
      <div class="p-name">${lastName}</div>
      <div class="p-pos">${pos.toUpperCase()}</div>
    </div>`;
  });
  document.getElementById('roster-grid').innerHTML = html || '<div style="grid-column:1/-1;text-align:center;color:var(--muted);font-family:var(--mono);font-size:0.7rem;padding:2rem;background:var(--panel);">No players found</div>';
}

// ─── TICKER ───────────────────────────────────────────────────────────────────
function buildTicker(ngRows, resRows) {
  const wins   = resRows.filter(r => parseInt(r.team_score||0) > parseInt(r.opponent_score||0)).length;
  const losses = resRows.length - wins;
  const pct    = resRows.length ? Math.round(wins/resRows.length*100)+'%' : '—';
  const opp    = ngRows.length ? (ngRows[0].opponent || '—').toUpperCase() : '—';
  const ha     = ngRows.length ? (ngRows[0].home_away || '').toUpperCase() : '';
  const venue  = ngRows.length ? (ngRows[0].venue || '').toUpperCase() : '';

  const items = [
    `AIR BALLERS <span class="ticker-sep">·</span> THESSALONIKI AMATEUR LEAGUE`,
    `SEASON 2025–26 <span class="ticker-sep">·</span> RECORD ${wins}W – ${losses}L`,
    `NEXT GAME <span class="ticker-sep">·</span> VS ${opp} <span class="ticker-sep">·</span> ${ha} @ ${venue}`,
    `WIN PCT ${pct}`,
  ];
  // duplicate for seamless loop
  const doubled = [...items, ...items].map(t => `<span class="ticker-item">${t}</span>`).join('');
  document.getElementById('ticker-track').innerHTML = doubled;
}
